Сортировка массива в С++

Сортировка массива методом выбора

int main()
{
	int arr[5]{ 30, 50, 20, 10, 40 };
	for (int i = 0; i < 5; i++)
	{
		for (int j = 0; j < 4; j++)
		{
			int var = arr[i];
			if (arr[i] < arr[j])
			{
				arr[i] = arr[j];
				arr[j] = var;
			}
		}
	}

	for(int i = 0; i < 5; i++) { cout << arr[i] << " "; }


	return 0;
}
